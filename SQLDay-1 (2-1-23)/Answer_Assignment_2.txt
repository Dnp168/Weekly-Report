PRIMARY KEY (studentid),
FOREIGN KEY(recid_1) REFERENCES Student_darshan(recid));

3)CREATE TABLE studentpresent_darshan 
(recid_2 int not null,

 Month int, 
present int, 
FOREIGN KEY(recid_2) REFERENCES Student_darshan(recid), 
FOREIGN KEY(studentid_1) REFERENCES Studentdetail_darshan(studentid));

4)CREATE TABLE studenthours_darshan
(recid_3 int not null, 
studentid_2 int not null,
 Datex Date, 
Hours int, 
FOREIGN KEY(recid_3) REFERENCES Student_darshan(recid), 
FOREIGN KEY(studentid_2) REFERENCES Studentdetail_darshan(studentid));

Write queries for the following requirements.

1)Display following details for each student:
   
        -   Name
        -   Standard
        -   Sex

Ans : select Name, standard, flagsex from 
Student_darshan, Studentdetail_darshan 
where 
Student_darshan.recid=Studentdetail_darshan.recid_1 order by Name asc;

2)Find SUM of student hours.
Ans: select Name, sum(hours) as Total_Hours from Student_darshan, studenthours_darshan where Student_darshan.recid=studenthours_darshan.recid_3 group by recid;

3)Find Average Present of each student.
Ans: select Name, avg(present) as Avg_Present from Student_darshan, studentpresent_darshan where Student_darshan.recid=studentpresent_darshan.recid_2 group by recid_2;

4)Find Minimum present among all students.
Ans: select Name, min(present) as Minimum_Present from Student_darshan, studentpresent_darshan where Student_darshan.recid=studentpresent_darshan.recid_2 group by recid_2;

5)Find Maximum present among all students.
Ans: select Name, max(present) as Maximum_Present from Student_darshan, studentpresent_darshan where Student_darshan.recid=studentpresent_darshan.recid_2 group by recid_2;


6)Display fields like Name, Date, Hours from "studenthours" table where Date should be in following format :


1)Date Formate : YYYY-MM-DD
select Name, Datex, Hours from Student_darshan, studenthours_darshan where Student_darshan.recid=studenthours_darshan.recid_3;

2)Date Formate : DD/MM/YY
select Name, DATE_FORMAT(Datex,'%d/%m/%Y') as Datex, Hours from Student_darshan, studenthours_darshan where Student_darshan.recid=studenthours_darshan.recid_3;
