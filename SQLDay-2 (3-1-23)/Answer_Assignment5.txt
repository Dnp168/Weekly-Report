Q-1 Write a query which will print name of each student, and his or her attendance report for current month. 
Fields : Name, Attendance Report for Current Month

	Under "Attendance Report for Current Month" column:
		if 'studentpresent.present' is less than or equal to 7 print "Poor", 
		if 'studentpresent.present' is more than 7 and less than or equals 15 print "Good", 
		if 'studentpresent.present' is more than 15 and less than or equals 20 print "Better", 
		if 'studentpresent.present' is more than 20 print "Best".

=>select Name,
CASE
                  When present>20 then 'Best'
                  When present>15 then 'Good'
                  When present>7 then 'Better'
                  Else 'Poor'
END AS Attendance_Report
from Student_darshan natural join studentpresent_darshan;

Q-3 Print information for the student having name like "Mira", "Kabir", "Om", "Manas"
	Fields: Name, Standard.
=> select Name, standard From Student_darshan natural join Studentdetail_darshan where Name in ('Jay','Mehul');

Q-4 Print information for the student having name other than "Mira", "Kabir", "Om", "Manas"
	Fields: Name, Standard
=> select Name, standard From Student_darshan natural join Studentdetail_darshan where Name not in ('Jay','Mehul');

Q-5 Print information for the student having presence for Jananuary between 20 to 25.
	Fields: Name, Standard, Present
=> select Name,standard,persent from Student_darshan as st inner join studentpresent_darshan as sp
    on st.recid = sp.recid
    where sp.Month=6 and sp.present between 15 and 20 and
    st.recid in (select recid, standard from Studentdetail_darshan);

Q-6 Print information for the student having charcter "e" at the second position of their name.
	Fields: Name, Standard
=> select Name, standard from Student_darshan natural join Studentdetail_darshan where Name like '_e%';

Q-7 Print information for the student having "bhai" in their name.

Q-8 Print information for the student who do not have any presence in the college.
	Fields: Name, Standard
=> select Name, standard from Student_darshan as st inner join Studentdetail_darshan as sd on st.recid=sd.recid
   inner join studentpresent_darshan as sp on st.recid=sp.recid where present<=20;
