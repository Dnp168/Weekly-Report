Q-1 Print total number of boys and girls within a single record.
	Fields : Boys, Girls

	Required Output:
	================
	Boys	Girls
	5	3

=> select
  count(case when flagsex='Male' then 1 end) as Boys,
  count(case when flagsex='Female' then 1 end) as Girls
  from Studentdetail_darshan;

Q-2 Find total number of Hours for Boys and Girls. 
	Fields : Gender, Present(Hrs)

	Required Output:
	================
	Gender	Present(Hrs)
	Boys	100
	Girls	120

=> select flagsex as Gender, sum(hours) from Studentdetail_darshan as sd 
   inner join 
   studenthours_darshan as sh on sd.recid=sh.recid 
   group by sd.flagsex;

Q-3 Find all latest prices for each code for each product.
	Fields : Product Name, Code, Date, Price

	Required Output:
	================
	Product Name	Code	Date		Price
	Computer	COMP	2007-01-01	Rs. 24,000/-

=> select pname as Product_Name, 
code as Product_Code, 
effdt as Date, 
price as Price 
from Product_darshan 
natural join 
productprice_darshan 
where 
effdt in 
(select max(effdt) 
from productprice_darshan group by code);
